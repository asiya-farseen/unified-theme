Note: This is a static HTML example showing how Unified Theme works with React components. In a real React app, you would import the CSS and use the classes directly on your JSX elements.

React Button Component

Here's how you might create a Button component in React using Unified Theme classes: 

// Button.jsx
import React from 'react';

const Button = ({ 
  children, 
  variant = 'primary', 
  size = 'md', 
  disabled = false, 
  onClick,
  ...props 
}) => {
  const baseClasses = 'btn';
  const variantClass = `btn-${variant}`;
  const sizeClass = `btn-${size}`;
  
  const className = [
    baseClasses,
    variantClass,
    sizeClass,
    disabled && 'disabled'
  ].filter(Boolean).join(' ');
  
  return (
    <button 
      className={className}
      disabled={disabled}
      onClick={onClick}
      {...props}
    >
      {children}
    </button>
  );
};

export default Button;

React Form Components

// Input.jsx
import React from 'react';

const Input = ({ 
  label, 
  error, 
  helpText, 
  size = 'md',
  ...props 
}) => {
  const inputClass = [
    'form-control',
    size !== 'md' && `form-control-${size}`,
    error && 'form-control-error'
  ].filter(Boolean).join(' ');
  
  return (
    <div className="form-group">
      {label && (
        <label className="form-label">{label}</label>
      )}
      <input className={inputClass} {...props} />
      {error && (
        <div className="text-error-600 text-sm mt-1">{error}</div>
      )}
      {helpText && (
        <div className="form-text">{helpText}</div>
      )}
    </div>
  );
};

React Card Component

// Card.jsx
import React from 'react';

const Card = ({ 
  children, 
  variant = 'default', 
  size = 'md',
  interactive = false,
  className = '',
  ...props 
}) => {
  const cardClass = [
    'card',
    variant !== 'default' && `card-${variant}`,
    size !== 'md' && `card-${size}`,
    interactive && 'card-interactive',
    className
  ].filter(Boolean).join(' ');
  
  return (
    <div className={cardClass} {...props}>
      {children}
    </div>
  );
};

const CardHeader = ({ children, className = '', ...props }) => (
  <div className={`card-header ${className}`} {...props}>
    {children}
  </div>
);

const CardBody = ({ children, className = '', ...props }) => (
  <div className={`card-body ${className}`} {...props}>
    {children}
  </div>
);

const CardTitle = ({ children, className = '', ...props }) => (
  <h3 className={`card-title ${className}`} {...props}>
    {children}
  </h3>
);

Card.Header = CardHeader;
Card.Body = CardBody;
Card.Title = CardTitle;

export default Card;

Integration Tips

CSS-in-JS: Unified Theme works great with styled-components, Emotion, and other CSS-in-JS libraries. Simply apply the classes using the className prop.
Tailwind CSS: Unified Theme is designed to complement Tailwind CSS. You can use both systems together seamlessly.
TypeScript: Create type definitions for your component props to ensure type safety when using Unified Theme classes.
Bundle Size: Import only the CSS you need by using the individual component stylesheets instead of the full bundle.
