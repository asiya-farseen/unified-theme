Vue.js Integration Example

Note: This is a static HTML example showing how Unified Theme works with Vue.js components. In a real Vue app, you would import the CSS and use the classes directly in your templates.

Vue Button Component

Here's how you might create a Button component in Vue.js using Unified Theme classes: 

<!-- Button.vue -->
<template>
  <button 
    :class="buttonClasses" 
    :disabled="disabled"
    @click="handleClick"
  >
    <slot />
  </button>
</template>

<script>
export default {
  name: 'Button',
  props: {
    variant: {
      type: String,
      default: 'primary'
    },
    size: {
      type: String,
      default: 'md'
    },
    disabled: {
      type: Boolean,
      default: false
    }
  },
  computed: {
    buttonClasses() {
      return [
        'btn',
        `btn-${this.variant}`,
        `btn-${this.size}`,
        {
          'disabled': this.disabled
        }
      ];
    }
  },
  methods: {
    handleClick(event) {
      if (!this.disabled) {
        this.$emit('click', event);
      }
    }
  }
};
</script>

Vue Form Component

<!-- FormInput.vue -->
<template>
  <div class="form-group">
    <label v-if="label" class="form-label">{{ label }}</label>
    <input 
      :class="inputClasses"
      :type="type"
      :placeholder="placeholder"
      :value="modelValue"
      @input="$emit('update:modelValue', $event.target.value)"
    />
    <div v-if="error" class="text-error-600 text-sm mt-1">
      {{ error }}
    </div>
    <div v-if="helpText" class="form-text">
      {{ helpText }}
    </div>
  </div>
</template>

<script>
export default {
  name: 'FormInput',
  props: {
    label: String,
    type: {
      type: String,
      default: 'text'
    },
    placeholder: String,
    modelValue: String,
    error: String,
    helpText: String,
    size: {
      type: String,
      default: 'md'
    }
  },
  emits: ['update:modelValue'],
  computed: {
    inputClasses() {
      return [
        'form-control',
        this.size !== 'md' && `form-control-${this.size}`,
        this.error && 'form-control-error'
      ].filter(Boolean);
    }
  }
};
</script>

Vue Modal Component

<!-- Modal.vue -->
<template>
  <div 
    v-if="show" 
    class="fixed inset-0 z-50 overflow-y-auto"
    @click="handleBackdropClick"
  >
    <div class="flex items-center justify-center min-h-screen px-4">
      <div class="fixed inset-0 bg-black opacity-50"></div>
      <div 
        class="relative bg-white rounded-lg shadow-lg max-w-md w-full"
        @click.stop
      >
        <div class="card">
          <div v-if="title" class="card-header card-header-bordered">
            <h3 class="card-title">{{ title }}</h3>
            <button 
              class="absolute top-4 right-4 text-neutral-400 hover:text-neutral-600"
              @click="close"
            >
              ×
            </button>
          </div>
          <div class="card-body">
            <slot />
          </div>
          <div v-if="$slots.footer" class="card-footer card-footer-bordered">
            <slot name="footer" />
          </div>
        </div>
      </div>
    </div>
  </div>
</template>

<script>
export default {
  name: 'Modal',
  props: {
    show: {
      type: Boolean,
      default: false
    },
    title: String,
    closable: {
      type: Boolean,
      default: true
    }
  },
  methods: {
    close() {
      if (this.closable) {
        this.$emit('close');
      }
    },
    handleBackdropClick() {
      if (this.closable) {
        this.close();
      }
    }
  }
};
</script>

Vue 3 Composition API

<!-- Counter.vue -->
<template>
  <div class="card text-center">
    <div class="card-body">
      <h3 class="card-title text-4xl font-bold text-primary-600 mb-4">
        {{ count }}
      </h3>
      <div class="flex justify-center gap-3">
        <button 
          class="btn btn-outline-primary"
          @click="decrement"
          :disabled="count <= 0"
        >
          -
        </button>
        <button class="btn btn-primary" @click="increment">
          +
        </button>
      </div>
    </div>
  </div>
</template>

<script setup>
import { ref, computed } from 'vue';

const count = ref(0);

const increment = () => {
  count.value++;
};

const decrement = () => {
  if (count.value > 0) {
    count.value--;
  }
};

const isZero = computed(() => count.value === 0);
</script>
